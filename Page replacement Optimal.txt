#include <stdio.h>
#define INF 9999

int main() {
    int frames, n, i, j, k, pos, pageFault = 0, flag, maxDist, dist;
    printf("Enter number of frames: ");
    scanf("%d", &frames);
    printf("Enter number of pages: ");
    scanf("%d", &n);
    int pages[n], f[frames];
    for(i=0; i<frames; i++) f[i] = -1;

    printf("Enter page reference string:\n");
    for(i=0; i<n; i++) scanf("%d", &pages[i]);

    printf("\nPage\tFrames\t\tHit/Fault\n");
    for(i=0; i<n; i++) {
        flag = 0;
        // Check if page hit
        for(j=0; j<frames; j++) {
            if(f[j] == pages[i]) {
                flag = 1; break;
            }
        }
        if(flag == 0) {
            // Page fault, find page to replace
            pos = -1; maxDist = -1;
            for(j=0; j<frames; j++) {
                if(f[j] == -1) { // empty frame found
                    pos = j; break;
                } else {
                    // Find next use distance
                    dist = INF;
                    for(k=i+1; k<n; k++)
                        if(pages[k] == f[j]) { dist = k; break; }
                    if(dist > maxDist) { maxDist = dist; pos = j; }
                }
            }
            f[pos] = pages[i];
            pageFault++;
        }
        printf("%d\t", pages[i]);
        for(j=0; j<frames; j++) {
            if(f[j] != -1) printf("%d ", f[j]);
            else printf("- ");
        }
        printf("\t%s\n", flag ? "Hit" : "Fault");
    }
    printf("Total page faults: %d\n", pageFault);
return 0;
}
